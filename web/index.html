            Шпаргалка: типы шагов и cron
            

Шаги:
  parse_sources  — собрать новости из источников. params: {"source_group":"all|rss|telethon|web","days":1,"limit":200}
  filter         — отсев. params: {"keywords_include":["ai"],"keywords_exclude":[],"min_text_length":30,"days":1}
  dedup          — LLM-дедупликация (активный промпт «dedup»)
  summary        — LLM-сводка. params: {"period_label":"сутки"}
  compose_post   — LLM-пост. params: {"prompt_name":"default","include_image":true,"instruction":""}
  publish        — отправка. params: {"bot_id":1,"chat_id":"@my_channel","parse_mode":"HTML","include_image":true}
  wait           — пауза. params: {"seconds":30}

Cron (5 полей: мин час день месяц день_недели):
  */15 * * * *    — каждые 15 минут
  0 9 * * 1-5     — будни в 9:00
  0 8,20 * * *    — 8 утра и 8 вечера
  0 */2 * * *     — каждые 2 часа
